/*
Light Column Variations
By ChronoSeth

Set the first argument on the actor to 1, 2, or 3 to change the light column's behaviour.
Arg = 1: Flickering light (mostly on)
Arg = 2: Flickering light (mostly off)
Arg = 3: Flashing light
Arg = 4: Off
*/


Actor NewColumnBase : Column //Base definition for below actors, also used for flickering/flashing/dead standard column
{
	//$Category Light Sources
	States
	{
		Spawn:
			COLU B 8 Bright
		Spawn2:
			COLU B 0 Bright A_JumpIf (Args[0] == 1, "Flicker")
			COLU B 0 A_JumpIf (Args[0] == 2, "Flicker2")
			COLU B 0 A_JumpIf (Args[0] == 3, "Flash")
			COLU B 0 A_JumpIf (Args[0] == 4, "Spawn3")
			Goto Spawn
		Spawn3:
			COLU B 8
			COLU B 0 A_JumpIf (Args[0] == 2, "Flicker2")
			Loop
		Flicker:
			COLU B 1 A_Jump (128, "Spawn")
			Loop
		Flicker2:
			COLU C 1 Bright A_Jump (128, "Spawn3")
			Loop
		Flash:
			COLU B 8
			Goto Spawn
	}
}
